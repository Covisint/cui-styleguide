The off canvas menu component requires two menus to be implemented in markup:

  1. The mobile menu wrapped in Snap.js specific classes
  2. The desktop menu wrapped in with the .cui-off-canvas class

The markup example below contains a few very important elements with classes that are necessary in order to enable the off canvas' functionality

  • .snap-drawers - This is where you put your markup for your mobile off-canvas menu
  • .snap-drawer - in our case, we use a single drawer
  • .cui-off-canvas - this block-level class, its elements and modifiers are used to apply style to the navigation menu
  • the snap-content element with .snap-content - this is where the content of your page should go that should be "pushed" when the mobile menu is expanded, including your header
  • .cui-menu__toggle-button-container - applied to the button used to open and close the menu
  • .cui-menu__content-wrapper - this div wraps your main content that should be shifted on the desktop. Should not contain items you don't want shifted, like your header element contents
  • .desktop-menu - applied to the nav element for your desktop menu

In addition to these items, the small but necessary Javascript snippet located below also must be included at the bottom of your page

JAVASCRIPT: 
                  
var Nav = (function () {

var cache = new Array();
cache.push(document.querySelector('.desktop-menu'));
cache.push(document.querySelector('.cui-menu__content-wrapper'));
cache.push(document.querySelector('.cui-menu__toggle-button-container'));

function toggleDesktopNav() {
  cache[0].classList.toggle('desktop-menu--collapse');
  cache[1].classList.toggle('desktop-menu--collapse');
  cache[2].classList.toggle('active');
};

return {
  navToggle: toggleDesktopNav
};

}) ();

// Snap.js Mobile Navigation

var snapMenu = new Snap({
    element: document.querySelector('.snap-content'),
    disable: 'right'
});

var menuToggle = document.querySelector('.cui-menu__toggle-button-container')

menuToggle.addEventListener('click', function() {

    if (snapMenu.state().state == "left") {
        snapMenu.close();
    } else {
        snapMenu.open('left');
    }

});
